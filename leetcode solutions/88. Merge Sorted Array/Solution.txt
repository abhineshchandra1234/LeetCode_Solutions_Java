//88. Merge Sorted Array



//kotlin
class Solution {
    fun merge(nums1: IntArray, m: Int, nums2: IntArray, n: Int) {
        var index = nums1.size - 1
        var i = m - 1
        var j = n - 1

        while (i >= 0 && j >= 0) {
            if (nums1[i] >= nums2[j]) {
                nums1[index--] = nums1[i--]
            } else {
                nums1[index--] = nums2[j--]
            }
        }

        while (j >= 0) {
            nums1[index--] = nums2[j--]
        }
    }
}
